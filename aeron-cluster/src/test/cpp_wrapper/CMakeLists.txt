endif ()

set(TEST_HEADERS
    ${AERON_CLIENT_WRAPPER_TEST_PATH}/EmbeddedMediaDriver.h
    ${AERON_CLIENT_WRAPPER_TEST_PATH}/TestUtil.h)

include_directories(${AERON_CLUSTER_WRAPPER_SOURCE_PATH})
include_directories(${AERON_CLIENT_WRAPPER_SOURCE_PATH})
include_directories(${AERON_CLIENT_WRAPPER_TEST_PATH})

# Add generated codec directory - this matches the binary directory from main CMakeLists.txt
# The include path should be the binary directory, not the generated subdirectory
# because code uses: #include "generated/aeron_cluster_codecs/EventCode.h"
set(CLUSTER_CODEC_BINARY_DIR "${CMAKE_BINARY_DIR}/aeron-cluster/src/main/cpp_wrapper")
include_directories(${CLUSTER_CODEC_BINARY_DIR})

# Add archive C++ codecs for tests that need them (e.g., EgressPollerTest)
# Archive C++ codecs are generated by aeron-archive's main CMakeLists.txt
# The include path should be the binary directory, not the generated subdirectory
# because code uses: #include "generated/aeron_archive_client/MessageHeader.h"
set(ARCHIVE_CODEC_BINARY_DIR "${CMAKE_BINARY_DIR}/aeron-archive/src/main/cpp_wrapper")

# Add archive codec include directory
include_directories(${ARCHIVE_CODEC_BINARY_DIR})

function(aeron_cluster_wrapper_test name file)
    set(wrapper_name "${name}W")
    add_executable(${wrapper_name} ${file} ${TEST_HEADERS})
    add_dependencies(${wrapper_name} gmock aeron_static aeron_driver_static cluster_cpp_codecs archive_cpp_codecs)
    # Link to aeron_cluster_wrapper first to inherit its include directories (including generated codecs)
    target_link_libraries(${wrapper_name} 
        aeron_cluster_wrapper 
        aeron_client_wrapper 
        aeron_archive_wrapper
        aeron_static 
        aeron_driver_static 
        gmock_main 
        ${CMAKE_THREAD_LIBS_INIT} 
        ${AERON_LIB_WINSOCK_LIBS})
    target_compile_definitions(${wrapper_name} PUBLIC "_SILENCE_TR1_NAMESPACE_DEPRECATION_WARNING")
    
    # Add debug symbols and sanitizers for better crash debugging
    if (NOT "${CMAKE_SYSTEM_NAME}" MATCHES "Windows")
        target_compile_options(${wrapper_name} PRIVATE
            -g
            -O0
            -fno-omit-frame-pointer
            -fno-inline
        )
        target_link_options(${wrapper_name} PRIVATE
            -g
            -fno-omit-frame-pointer
        )
    endif()
    
    add_test(NAME ${wrapper_name} COMMAND ${wrapper_name})
endfunction()
